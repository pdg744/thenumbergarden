\newcommand{\businesscardpage}{%
  \begin{tikzpicture}[remember picture,overlay,x=1in,y=1in]
    \coordinate (pageNW) at ([xshift=0.5in,yshift=-0.125in] current page.north west);
    \foreach \col in {0,1} {
      \foreach \row in {0,1,2,3,4} {
        \pgfmathsetmacro{\x}{\col * 3.75}
        \pgfmathsetmacro{\y}{-\row * 2.1875}
        \drawbusinesscard{\x}{\y}
      }
    }
  \end{tikzpicture}%
  \null
}

\newcommand{\flyerpage}{%
  \begin{tikzpicture}[remember picture,overlay,x=1in,y=1in]
    \coordinate (flyerNW) at (current page.north west);
    \foreach \col in {0,1} {
      \foreach \row in {0,1} {
        \pgfmathsetmacro{\x}{\col * 4.25}
        \pgfmathsetmacro{\y}{-\row * 5.5}
        \drawflyer{\x}{\y}
      }
    }
  \end{tikzpicture}%
  \null
}
